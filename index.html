#!/usr/bin/env python3 """ assinar_pdf.py — Agente de Assinatura Automática de PDFs Uso: python3 assinar_pdf.py [saida.pdf] Funcionamento: 1. Detecta via OCR o campo "Assinatura do Favorecido" no PDF 2. Encontra a linha tracejada acima desse texto 3. Insere a imagem da assinatura RENTE à linha (sem espaço em branco) 4. A assinatura fica sobreposta sobre a linha, como uma assinatura real 5. Salva o PDF assinado Requisitos: pip install pypdf reportlab pdf2image pytesseract pillow numpy sudo apt-get install poppler-utils tesseract-ocr """ import sys import io import tempfile import argparse from pathlib import Path import numpy as np from PIL import Image from pdf2image import convert_from_path import pytesseract from pypdf import PdfReader, PdfWriter from reportlab.pdfgen import canvas # ───────────────────────────────────────────── # 1. Preparar imagem da assinatura # ───────────────────────────────────────────── def preparar_assinatura(sig_path: str) -> Image.Image: """ Converte a assinatura PNG (fundo preto, traços cinza) em RGBA com fundo transparente e traços escuros prontos para overlay no papel branco. Corta o bounding box justo ao redor dos traços. """ sig = Image.open(sig_path).convert('RGB') arr = np.array(sig).astype(float) brightness = np.max(arr, axis=2) # brilho máximo por pixel # Crop rente ao conteúdo (ignora margem preta) rows = np.where(np.any(brightness > 30, axis=1))[0] cols = np.where(np.any(brightness > 30, axis=0))[0] if len(rows) == 0 or len(cols) == 0: raise ValueError("Assinatura parece estar em branco ou toda preta.") pad = 5 r0 = max(0, rows[0] - pad) r1 = min(arr.shape[0], rows[-1] + 1 + pad) c0 = max(0, cols[0] - pad) c1 = min(arr.shape[1], cols[-1] + 1 + pad) arr_c = arr[r0:r1, c0:c1] bright_c = brightness[r0:r1, c0:c1] # Inverte cor: cinza claro (traço) → cinza escuro no papel branco rgba = np.zeros((arr_c.shape[0], arr_c.shape[1], 4), dtype=np.uint8) inv = (255 - arr_c).astype(np.uint8) rgba[:, :, 0] = inv[:, :, 0] rgba[:, :, 1] = inv[:, :, 1] rgba[:, :, 2] = inv[:, :, 2] rgba[:, :, 3] = bright_c.astype(np.uint8) # alpha = brilho original return Image.fromarray(rgba, 'RGBA') # ───────────────────────────────────────────── # 2. Localizar a linha de assinatura no PDF # ───────────────────────────────────────────── def encontrar_linha_assinatura(page_img: Image.Image): """ Usa OCR para achar 'Assinatura do Favorecido' e então detecta a linha tracejada imediatamente acima desse texto. Retorna (line_top, line_bottom, line_x_left, line_x_right) em pixels da imagem renderizada (150 dpi). Retorna None se não encontrar a linha. """ arr = np.array(page_img.convert('L')) # OCR para achar o texto âncora data = pytesseract.image_to_data(page_img, output_type=pytesseract.Output.DICT) sig_text_top = None for i, text in enumerate(data['text']): if 'ssinatura' in text.lower(): sig_text_top = data['top'][i] break if sig_text_top is None: return None # Procura linha tracejada nas 200px acima do texto search_start = max(0, sig_text_top - 200) search_end = sig_text_top line_rows = [ r for r in range(search_start, search_end) if np.sum(arr[r] < 150) > 50 ] if not line_rows: return None line_top = min(line_rows) line_bottom = max(line_rows) mid_row = arr[line_rows[len(line_rows) // 2]] dark_cols = np.where(mid_row < 150)[0] return line_top, line_bottom, int(dark_cols[0]), int(dark_cols[-1]) # ───────────────────────────────────────────── # 3. Assinar o PDF # ───────────────────────────────────────────── def assinar_pdf(input_pdf: str, sig_png: str, output_pdf: str, dpi: int = 150): print(f"\n📄 PDF de entrada : {input_pdf}") print(f"✍️ Assinatura : {sig_png}") print(f"💾 PDF de saída : {output_pdf}\n") # Prepara imagem da assinatura sig_img = preparar_assinatura(sig_png) reader = PdfReader(input_pdf) writer = PdfWriter() pages_images = convert_from_path(input_pdf, dpi=dpi) for page_num, (pdf_page, page_img) in enumerate(zip(reader.pages, pages_images)): img_w, img_h = page_img.size pdf_w = float(pdf_page.mediabox.width) pdf_h = float(pdf_page.mediabox.height) scale_x = pdf_w / img_w scale_y = pdf_h / img_h resultado = encontrar_linha_assinatura(page_img) if resultado is None: print(f" Página {page_num+1}: campo 'Assinatura do Favorecido' não encontrado — ignorado.") writer.add_page(pdf_page) continue line_top, line_bottom, line_x_left, line_x_right = resultado print(f" Página {page_num+1}: linha encontrada (linhas {line_top}–{line_bottom} px, " f"cols {line_x_left}–{line_x_right} px)") # Dimensões desejadas da assinatura no PDF (em pontos) line_width_pts = (line_x_right - line_x_left) * scale_x sig_orig_w, sig_orig_h = sig_img.size sig_scale = line_width_pts / sig_orig_w sig_w_pts = line_width_pts sig_h_pts = sig_orig_h * sig_scale # Posição: x alinhado com a linha, fundo da assinatura RENTE ao topo da linha # Em PDF/ReportLab, y=0 é a base da página (cresce para cima) # drawImage(x, y, w, h): y é o canto inferior esquerdo da imagem # Para a assinatura ficar rente ACIMA da linha: bottom da imagem = topo da linha line_top_from_bottom = pdf_h - (line_top * scale_y) sig_x = line_x_left * scale_x sig_y = line_top_from_bottom # img bottom = line top → assinatura fica acima da linha print(f" Posição PDF: x={sig_x:.1f}, y={sig_y:.1f}, w={sig_w_pts:.1f}, h={sig_h_pts:.1f} pts") # Salva sig temporariamente e cria overlay PDF with tempfile.NamedTemporaryFile(suffix='.png', delete=False) as tmp: sig_img.save(tmp.name) tmp_path = tmp.name try: overlay_buf = io.BytesIO() c = canvas.Canvas(overlay_buf, pagesize=(pdf_w, pdf_h)) c.drawImage(tmp_path, sig_x, sig_y, width=sig_w_pts, height=sig_h_pts, mask='auto') c.save() overlay_buf.seek(0) overlay_page = PdfReader(overlay_buf).pages[0] pdf_page.merge_page(overlay_page) finally: Path(tmp_path).unlink(missing_ok=True) writer.add_page(pdf_page) print(f" ✅ Assinatura aplicada com sucesso!") with open(output_pdf, 'wb') as f: writer.write(f) print(f"\n✅ PDF assinado salvo em: {output_pdf}") # ───────────────────────────────────────────── # 4. CLI # ───────────────────────────────────────────── def main(): parser = argparse.ArgumentParser( description='Assina automaticamente PDFs com campo "Assinatura do Favorecido".' ) parser.add_argument('pdf', help='Caminho do PDF de entrada') parser.add_argument('assinatura', help='Caminho do PNG da assinatura') parser.add_argument('saida', nargs='?', help='Caminho do PDF de saída (padrão: _assinado.pdf)') args = parser.parse_args() input_path = Path(args.pdf) if not input_path.exists(): print(f"Erro: arquivo não encontrado: {args.pdf}") sys.exit(1) sig_path = Path(args.assinatura) if not sig_path.exists(): print(f"Erro: assinatura não encontrada: {args.assinatura}") sys.exit(1) output_path = args.saida or str(input_path.parent / (input_path.stem + '_assinado.pdf')) assinar_pdf(str(input_path), str(sig_path), output_path) if __name__ == '__main__': main()
